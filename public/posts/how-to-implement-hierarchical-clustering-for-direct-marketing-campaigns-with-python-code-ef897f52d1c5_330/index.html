def remove_outliers(data):

  df = data.copy()

  for col in list(df.columns):
    Q1 = df[str(col)].quantile(0.05)
    Q3 = df[str(col)].quantile(0.95)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    df = df[(df[str(col)] >= lower_bound) & (df[str(col)] <= upper_bound)]

  return df
